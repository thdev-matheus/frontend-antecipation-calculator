# Result

O componente de resultados é bastante simples de entender, sendo o componente que recebe o conteúdo que veio da api e o renderiza na tela em formato de textos.

Vejamos o layout e o código:

<img src="https://i.ibb.co/k4x9QhM/Captura-de-tela-de-2022-11-18-21-10-19.png" />

```typescript
// src/components/REsult/index.tsx

import { useRequestAPI } from "../../contexts/RequestAPI";
import { BoxData, BoxTitle, Container } from "./styles";
import { convertCentsToReal } from "../../utils/convert";

export const Result = () => {
  // busca no contexto o conteúdo que veio direto da api
  const { calculatedData } = useRequestAPI();

  // depois verifica se existem os campos e renderiza na tela em format o de texto
  return (
    <Container>
      <BoxTitle>
        <h3>Você receberá:</h3>
      </BoxTitle>

      <BoxData>
        {!!calculatedData[1] && (
          <span>
            {`Amanhã: R$ ${convertCentsToReal(calculatedData[1]).toFixed(2)}`}
          </span>
        )}
        {!!calculatedData[15] && (
          <span>
            {`Em 15 dias: R$ ${convertCentsToReal(calculatedData[15]).toFixed(
              2
            )}`}
          </span>
        )}
        {!!calculatedData[30] && (
          <span>
            {`Em 30 dias: R$ ${convertCentsToReal(calculatedData[30]).toFixed(
              2
            )}`}
          </span>
        )}
        {!!calculatedData[90] && (
          <span>
            {`Em 90 dias: R$ ${convertCentsToReal(calculatedData[90]).toFixed(
              2
            )}`}
          </span>
        )}
        {!!calculatedData[180] && (
          <span>
            {`Em 180 dias: R$ ${convertCentsToReal(calculatedData[180]).toFixed(
              2
            )}`}
          </span>
        )}
      </BoxData>
    </Container>
  );
};
```

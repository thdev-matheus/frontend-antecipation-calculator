# Configuração de temas para dark/ligth mode

<p>
  Esses objetos são utilizados para fornecer a paleta de cores para a aplicação
  a depnder do contexto "Dark Mode".
</p>
<p>É instanciado no arquivo src/app.tsx com uma lógica simples.</p>

```typescript
// src/themes/index.tsx
export const ligthTheme = {
  primary: "#fbfbfb",
  secondary: "#2d3436",
  error: "#D40404",
  darkBlue: "#0984e3",
  lightBlue: "#74b9ff",
  white: "#f9f9f9",
  black: "#151515",
  transparent: "#15151575",
};

export const DarkTheme = {
  primary: "#2d3436",
  secondary: "#fbfbfb",
  error: "#D40404",
  darkBlue: "#0984e3",
  lightBlue: "#74b9ff",
  white: "#f9f9f9",
  black: "#151515",
  transparent: "#15151575",
};

// ---------------------------------------------------------
// ---------------------------------------------------------

// src/app.tsx
import { RouterProvider } from "react-router-dom";
import { router } from "./routes";
import { ThemeProvider } from "styled-components";
import { DarkTheme, ligthTheme } from "./themes"; // 1 - importa-se os temas
import { useDarkMode } from "./contexts/DarkMode";

export const App = () => {
  const { isDarkMode } = useDarkMode(); // 2 - pega o valor do estado do contexto

  return (
    /* 3 - verifica se o estado está em dark mode ou não e fornece o tema de acordo com o resulado */
    <ThemeProvider theme={isDarkMode ? DarkTheme : ligthTheme}>
      <RouterProvider router={router} />
    </ThemeProvider>
  );
};
```
